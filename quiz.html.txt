<p>Do you like X?</p>
<button onclick="answer(1)">Yes</button>
<button onclick="answer(-1)">No</button>
<button onclick="answer(0)">I don't know</button>

<p id="result"></p>

<script>
let score = 0;

function answer(value) {
  score += value;

  if (score >= 2) {
    document.getElementById("result").innerText = "Result A";
  } else if (score >= 0) {
    document.getElementById("result").innerText = "Result B";
  } else {
    document.getElementById("result").innerText = "Result C";
  }
}
</script>
